   526	private fun DrawScope.drawFigmaImage(
   527	    image: ImageBitmap,
   528	    bounds: FigmaBounds,
   529	    stageLeft: Float,
   530	    stageTop: Float,
   531	    stageScale: Float,
   532	    alpha: Float = 1f,
   533	    clipCircle: Boolean = false
   534	) {
   535	    val left = stageLeft + bounds.x * stageScale
   536	    val top = stageTop + bounds.y * stageScale
   537	    val width = bounds.w * stageScale
   538	    val height = bounds.h * stageScale
   539	
   540	    if (clipCircle) {
   541	        val clip = Path().apply {
   542	            addOval(Rect(left, top, left + width, top + height))
   543	        }
   544	        clipPath(clip) {
   545	            drawImageBitmap(image, left, top, width, height, alpha)
   546	        }
   547	    } else {
   548	        drawImageBitmap(image, left, top, width, height, alpha)
   549	    }
   550	}

   319	    canvas.drawBitmap(overlay, 0f, 0f, paint)
   320	    return output
   321	}
   322	
   323	private fun circularBitmap(source: Bitmap): Bitmap {
   324	    val output = Bitmap.createBitmap(source.width, source.height, Bitmap.Config.ARGB_8888)
   325	    val canvas = Canvas(output)
   326	    val rect = RectF(0f, 0f, source.width.toFloat(), source.height.toFloat())
   327	    val maskPaint = Paint(Paint.ANTI_ALIAS_FLAG)
   328	    val imagePaint = Paint(Paint.ANTI_ALIAS_FLAG or Paint.FILTER_BITMAP_FLAG or Paint.DITHER_FLAG)
   329	    canvas.drawOval(rect, maskPaint)
   330	    imagePaint.xfermode = PorterDuffXfermode(PorterDuff.Mode.SRC_IN)
   331	    canvas.drawBitmap(source, null, rect, imagePaint)
   332	    imagePaint.xfermode = null
   333	    return output
   334	}

     1	# Cinerific
     2	
     3	Local Android project for the `prahlin/cinerific` GitHub repo.
     4	
     5	This is a first Kotlin / Jetpack Compose scaffold based on the Figma intro and sign-in frames. The current app opens as a tablet-friendly full-screen frame preview with:
     6	
     7	- Intro frame 1
     8	- Animated intro frame 2
     9	- Promo/logo intro frame 3
    10	- Sign-in/loading frame 4
    11	
    12	## Stack
    13	
    14	- Kotlin
    15	- Jetpack Compose Material 3
    16	- Android Gradle Plugin 8.5.x
    17	- Gradle 8.7
    18	
    19	## Run it
    20	
    21	Open the project in Android Studio and run `app`, or use the Gradle wrapper:
    22	
    23	```bash
    24	./gradlew :app:assembleDebug
    25	```
    26	
    27	The project pins Gradle to Android Studio's bundled JBR in `gradle.properties`, so command-line builds do not accidentally use an unsupported system JDK.
    28	
    29	From VS Code, run the default build task `Android: assemble debug`. There are also tasks for `Android: install debug` and `Android: launch Cinerific` when an emulator or device is connected.
    30	
    31	## Notes for the first Kotlin pass
    32	
    33	- The layout is intentionally simple and adaptive for tablets.
    34	- The current visuals are Compose approximations of the Figma frames and are meant to be replaced with production assets screen by screen.
    35	- The local project folder is the same path as the GitHub remote: `prahlin/cinerific`.
