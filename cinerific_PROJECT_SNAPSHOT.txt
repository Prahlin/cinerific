   162	@Composable
   163	private fun CinerificMainExperience(
   164	    signedInProfile: CinerificProfile,
   165	    selectedLanguage: CinerificLanguage,
   166	    onLanguageSelected: (CinerificLanguage) -> Unit,
   167	    onSignOut: () -> Unit
   168	) {
   169	    var destination by remember { mutableStateOf(CinerificDestination.Home) }
   170	    var selectedProgramTitle by rememberSaveable { mutableStateOf("Sink or Swim") }
   171	    var favoriteProgramTitles by rememberSaveable { mutableStateOf(emptyList<String>()) }
   172	    var favoritesFullPromptRequestId by remember { mutableStateOf(0) }
   173	    var autoLogoutEnabled by rememberSaveable { mutableStateOf(false) }
   174	    var userInitiatedPlaybackActive by remember { mutableStateOf(false) }
   175	    var lastInteractionMillis by remember { mutableStateOf(SystemClock.uptimeMillis()) }
   176	    val playbackSessionController = remember(userInitiatedPlaybackActive) {
   177	        CinerificPlaybackSessionController(
   178	            isUserInitiatedPlaybackActive = userInitiatedPlaybackActive,
   179	            onUserInitiatedPlaybackStarted = {
   180	                userInitiatedPlaybackActive = true
   181	            },
   182	            finishUserInitiatedPlayback = {
   183	                userInitiatedPlaybackActive = false
   184	                lastInteractionMillis = SystemClock.uptimeMillis()
   185	            }
   186	        )
   187	    }
   188	
   189	    LaunchedEffect(autoLogoutEnabled, userInitiatedPlaybackActive, lastInteractionMillis) {
   190	        if (!autoLogoutEnabled || userInitiatedPlaybackActive) return@LaunchedEffect
   191	        delay(AUTO_LOGOUT_TIMEOUT_MS)
   192	        val inactiveForMillis = SystemClock.uptimeMillis() - lastInteractionMillis
   193	        if (
   194	            autoLogoutEnabled &&
   195	            !userInitiatedPlaybackActive &&
   196	            inactiveForMillis >= AUTO_LOGOUT_TIMEOUT_MS
   197	        ) {
   198	            onSignOut()
   199	        }
   200	    }

   207	    fun toggleFavoriteProgram(title: String) {
   208	        if (title in favoriteProgramTitles) {
   209	            favoriteProgramTitles = favoriteProgramTitles - title
   210	            return
   211	        }
   212	
   213	        val isShow = cinerificProgramIsShow(title)
   214	        val filledSlots = cinerificFilledFavoritePlaceholderCount(
   215	            titles = favoriteProgramTitles,
   216	            isShow = isShow
   217	        )
   218	        if (filledSlots >= CINERIFIC_FAVORITES_CAPACITY) {
   219	            favoritesFullPromptRequestId += 1
   220	            return
   221	        }
   222	
   223	        favoriteProgramTitles = favoriteProgramTitles + title
   224	    }

   671	    alpha: Float = 1f,

   673	) {

     1	# Cinerific
     2	
     3	Local Android project for the `prahlin/cinerific` GitHub repo.
     4	
     5	This is a first Kotlin / Jetpack Compose scaffold based on the Figma intro and sign-in frames. The current app opens as a tablet-friendly full-screen frame preview with:
     6	
     7	- Intro frame 1
     8	- Animated intro frame 2
     9	- Promo/logo intro frame 3
    10	- Sign-in/loading frame 4
    11	
    12	## Stack
    13	
    14	- Kotlin
    15	- Jetpack Compose Material 3
    16	- Android Gradle Plugin 8.5.x
    17	- Gradle 8.7
    18	
    19	## Run it
    20	
    21	Open the project in Android Studio and run `app`, or use the Gradle wrapper:
    22	
    23	```bash
    24	./gradlew :app:assembleDebug
    25	```
    26	
    27	The project pins Gradle to Android Studio's bundled JBR in `gradle.properties`, so command-line builds do not accidentally use an unsupported system JDK.
    28	
    29	From VS Code, run the default build task `Android: assemble debug`. There are also tasks for `Android: install debug` and `Android: launch Cinerific` when an emulator or device is connected.
    30	
    31	## Notes for the first Kotlin pass
    32	
    33	- The layout is intentionally simple and adaptive for tablets.
    34	- The current visuals are Compose approximations of the Figma frames and are meant to be replaced with production assets screen by screen.
    35	- The local project folder is the same path as the GitHub remote: `prahlin/cinerific`.
