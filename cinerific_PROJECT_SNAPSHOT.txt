   648	private fun bootProgressAt(bootStartMillis: Long): Float {
   649	    val elapsed = SystemClock.uptimeMillis() - bootStartMillis
   650	    return (elapsed / BOOT_ANIMATION_MS.toFloat()).coerceIn(0f, 1f)

   653	private fun easedSegment(value: Float, start: Float, end: Float): Float {
   654	    val progress = ((value - start) / (end - start)).coerceIn(0f, 1f)
   655	    return FastOutSlowInEasing.transform(progress)
   656	}
   657	
   658	private fun linearSegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   659	    val ms = progress.coerceIn(0f, 1f) * BOOT_ANIMATION_MS
   660	    return ((ms - startMs) / (endMs - startMs).toFloat()).coerceIn(0f, 1f)
   661	}
   662	
   663	private fun easedSegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   664	    return FastOutSlowInEasing.transform(linearSegmentMs(progress, startMs, endMs))
   665	}
   666	
   667	private fun bouncySegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   668	    return bouncyEasing(linearSegmentMs(progress, startMs, endMs))
   669	}
   670	
   671	private fun bouncyEasing(value: Float): Float {
   672	    val t = value.coerceIn(0f, 1f)
   673	    val base = FastOutSlowInEasing.transform(t)
   674	    val bounce = sin(t * PI.toFloat() * 4.5f) * (1f - t) * 0.13f
   675	    return base + bounce
   676	}
   677	
   678	private fun lerpFloat(start: Float, end: Float, amount: Float) = start + (end - start) * amount

   683	    w = lerpFloat(start.w, end.w, amount),
   684	    h = lerpFloat(start.h, end.h, amount)

   156	        canvas: Canvas,
   157	        bitmap: Bitmap,
   158	        bounds: Bounds,
   159	        stageLeft: Float,
   160	        stageTop: Float,

   180	private data class Bounds(
   181	    val x: Float,
   182	    val y: Float,
   183	    val w: Float,
   184	    val h: Float
   185	)

   187	private fun bootProgressAt(bootStartMillis: Long): Float {
   188	    val elapsed = SystemClock.uptimeMillis() - bootStartMillis
   189	    return (elapsed / BOOT_ANIMATION_MS.toFloat()).coerceIn(0f, 1f)
   190	}
   191	
   192	private fun linearSegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   193	    val ms = progress.coerceIn(0f, 1f) * BOOT_ANIMATION_MS
   194	    return ((ms - startMs) / (endMs - startMs).toFloat()).coerceIn(0f, 1f)
   195	}
   196	
   197	private fun easedSegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   198	    return FastOutSlowInEasing.transform(linearSegmentMs(progress, startMs, endMs))
   199	}
   200	
   201	private fun bouncySegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   202	    return bouncyEasing(linearSegmentMs(progress, startMs, endMs))
   203	}
   204	
   205	private fun bouncyEasing(value: Float): Float {
   206	    val t = value.coerceIn(0f, 1f)
   207	    val base = FastOutSlowInEasing.transform(t)
   208	    val bounce = sin(t * PI.toFloat() * 4.5f) * (1f - t) * 0.13f
   209	    return base + bounce
   210	}
   211	
   212	private fun lerpFloat(start: Float, end: Float, amount: Float) = start + (end - start) * amount
   213	
   214	private fun lerpBounds(start: Bounds, end: Bounds, amount: Float) = Bounds(
   215	    x = lerpFloat(start.x, end.x, amount),
   216	    y = lerpFloat(start.y, end.y, amount),
   217	    w = lerpFloat(start.w, end.w, amount),
   218	    h = lerpFloat(start.h, end.h, amount)
   219	)
   220	
   221	private fun lerpColor(start: Int, end: Int, amount: Float): Int {
   222	    val t = amount.coerceIn(0f, 1f)
   223	    return Color.argb(
   224	        lerpInt(Color.alpha(start), Color.alpha(end), t),
   225	        lerpInt(Color.red(start), Color.red(end), t),
   226	        lerpInt(Color.green(start), Color.green(end), t),
   227	        lerpInt(Color.blue(start), Color.blue(end), t)
   228	    )
   229	}
   230	
   231	private fun lerpInt(start: Int, end: Int, amount: Float): Int {
   232	    return (start + (end - start) * amount).roundToInt()
   233	}
   234	
   235	private fun combineBitmaps(base: Bitmap, overlay: Bitmap): Bitmap {
   236	    val output = Bitmap.createBitmap(base.width, base.height, Bitmap.Config.ARGB_8888)
   237	    val canvas = Canvas(output)
   238	    val paint = Paint(Paint.ANTI_ALIAS_FLAG or Paint.FILTER_BITMAP_FLAG or Paint.DITHER_FLAG)
   239	    canvas.drawBitmap(base, 0f, 0f, paint)
   240	    canvas.drawBitmap(overlay, 0f, 0f, paint)
   241	    return output
   242	}
   243	
   244	private fun circularBitmap(source: Bitmap): Bitmap {
   245	    val output = Bitmap.createBitmap(source.width, source.height, Bitmap.Config.ARGB_8888)
   246	    val canvas = Canvas(output)
   247	    val rect = RectF(0f, 0f, source.width.toFloat(), source.height.toFloat())
   248	    val maskPaint = Paint(Paint.ANTI_ALIAS_FLAG)
   249	    val imagePaint = Paint(Paint.ANTI_ALIAS_FLAG or Paint.FILTER_BITMAP_FLAG or Paint.DITHER_FLAG)
   250	    canvas.drawOval(rect, maskPaint)
   251	    imagePaint.xfermode = PorterDuffXfermode(PorterDuff.Mode.SRC_IN)
   252	    canvas.drawBitmap(source, null, rect, imagePaint)
   253	    imagePaint.xfermode = null
   254	    return output
   255	}

     1	# Cinerific
     2	
     3	Local Android project for the `prahlin/cinerific` GitHub repo.
     4	
     5	This is a first Kotlin / Jetpack Compose scaffold based on the Figma intro and sign-in frames. The current app opens as a tablet-friendly full-screen frame preview with:
     6	
     7	- Intro frame 1
     8	- Animated intro frame 2
     9	- Promo/logo intro frame 3
    10	- Sign-in/loading frame 4
    11	
    12	## Stack
    13	
    14	- Kotlin
    15	- Jetpack Compose Material 3
    16	- Android Gradle Plugin 8.5.x
    17	- Gradle 8.7
    18	
    19	## Run it
    20	
    21	Open the project in Android Studio and run `app`, or use the Gradle wrapper:
    22	
    23	```bash
    24	./gradlew :app:assembleDebug
    25	```
    26	
    27	The project pins Gradle to Android Studio's bundled JBR in `gradle.properties`, so command-line builds do not accidentally use an unsupported system JDK.
    28	
    29	From VS Code, run the default build task `Android: assemble debug`. There are also tasks for `Android: install debug` and `Android: launch Cinerific` when an emulator or device is connected.
    30	
    31	## Notes for the first Kotlin pass
    32	
    33	- The layout is intentionally simple and adaptive for tablets.
    34	- The current visuals are Compose approximations of the Figma frames and are meant to be replaced with production assets screen by screen.
    35	- The local project folder is the same path as the GitHub remote: `prahlin/cinerific`.
