   680	private fun easedSegment(value: Float, start: Float, end: Float): Float {
   681	    val progress = ((value - start) / (end - start)).coerceIn(0f, 1f)
   682	    return FastOutSlowInEasing.transform(progress)
   683	}

   685	private fun linearSegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   686	    val ms = progress.coerceIn(0f, 1f) * BOOT_ANIMATION_MS
   687	    return ((ms - startMs) / (endMs - startMs).toFloat()).coerceIn(0f, 1f)
   688	}
   689	
   690	private fun easedSegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   691	    return FastOutSlowInEasing.transform(linearSegmentMs(progress, startMs, endMs))
   692	}

   694	private fun bouncySegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   695	    return bouncyEasing(linearSegmentMs(progress, startMs, endMs))
   696	}
   697	
   698	private fun bouncyEasing(value: Float): Float {
   699	    val t = value.coerceIn(0f, 1f)
   700	    val base = FastOutSlowInEasing.transform(t)
   701	    val bounce = sin(t * PI.toFloat() * 4.5f) * (1f - t) * 0.13f
   702	    return base + bounce
   703	}

   230	        if (!isFinalFrameSettled() || width <= 0 || height <= 0) return false
   231	
   232	        val stageScale = min(width / FIGMA_FRAME_WIDTH, height / FIGMA_FRAME_HEIGHT)
   233	        val stageLeft = (width - FIGMA_FRAME_WIDTH * stageScale) / 2f
   234	        val stageTop = (height - FIGMA_FRAME_HEIGHT * stageScale) / 2f
   235

   237	            val centerX = stageLeft + (bounds.x + bounds.w / 2f) * stageScale
   238	            val centerY = stageTop + (bounds.y + bounds.h / 2f) * stageScale
   239	            val radius = min(bounds.w, bounds.h) * stageScale / 2f
   240	            val dx = x - centerX
   241	            val dy = y - centerY
   242	            dx * dx + dy * dy <= radius * radius

   254	private fun bootProgressAt(bootStartMillis: Long): Float {
   255	    val elapsed = SystemClock.uptimeMillis() - bootStartMillis
   256	    return (elapsed / BOOT_ANIMATION_MS.toFloat()).coerceIn(0f, 1f)
   257	}
   258	
   259	private fun linearSegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   260	    val ms = progress.coerceIn(0f, 1f) * BOOT_ANIMATION_MS
   261	    return ((ms - startMs) / (endMs - startMs).toFloat()).coerceIn(0f, 1f)
   262	}
   263	
   264	private fun easedSegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   265	    return FastOutSlowInEasing.transform(linearSegmentMs(progress, startMs, endMs))
   266	}
   267	
   268	private fun bouncySegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   269	    return bouncyEasing(linearSegmentMs(progress, startMs, endMs))
   270	}
   271	
   272	private fun bouncyEasing(value: Float): Float {
   273	    val t = value.coerceIn(0f, 1f)
   274	    val base = FastOutSlowInEasing.transform(t)
   275	    val bounce = sin(t * PI.toFloat() * 4.5f) * (1f - t) * 0.13f
   276	    return base + bounce
   277	}
   278	
   279	private fun lerpFloat(start: Float, end: Float, amount: Float) = start + (end - start) * amount
   280	
   281	private fun lerpBounds(start: Bounds, end: Bounds, amount: Float) = Bounds(
   282	    x = lerpFloat(start.x, end.x, amount),
   283	    y = lerpFloat(start.y, end.y, amount),
   284	    w = lerpFloat(start.w, end.w, amount),
   285	    h = lerpFloat(start.h, end.h, amount)
   286	)
   287	
   288	private fun lerpColor(start: Int, end: Int, amount: Float): Int {
   289	    val t = amount.coerceIn(0f, 1f)
   290	    return Color.argb(
   291	        lerpInt(Color.alpha(start), Color.alpha(end), t),
   292	        lerpInt(Color.red(start), Color.red(end), t),
   293	        lerpInt(Color.green(start), Color.green(end), t),
   294	        lerpInt(Color.blue(start), Color.blue(end), t)
   295	    )
   296	}
   297	
   298	private fun lerpInt(start: Int, end: Int, amount: Float): Int {
   299	    return (start + (end - start) * amount).roundToInt()
   300	}
   301	
   302	private fun combineBitmaps(base: Bitmap, overlay: Bitmap): Bitmap {
   303	    val output = Bitmap.createBitmap(base.width, base.height, Bitmap.Config.ARGB_8888)
   304	    val canvas = Canvas(output)
   305	    val paint = Paint(Paint.ANTI_ALIAS_FLAG or Paint.FILTER_BITMAP_FLAG or Paint.DITHER_FLAG)
   306	    canvas.drawBitmap(base, 0f, 0f, paint)
   307	    canvas.drawBitmap(overlay, 0f, 0f, paint)
   308	    return output
   309	}
   310	
   311	private fun circularBitmap(source: Bitmap): Bitmap {
   312	    val output = Bitmap.createBitmap(source.width, source.height, Bitmap.Config.ARGB_8888)
   313	    val canvas = Canvas(output)
   314	    val rect = RectF(0f, 0f, source.width.toFloat(), source.height.toFloat())
   315	    val maskPaint = Paint(Paint.ANTI_ALIAS_FLAG)
   316	    val imagePaint = Paint(Paint.ANTI_ALIAS_FLAG or Paint.FILTER_BITMAP_FLAG or Paint.DITHER_FLAG)
   317	    canvas.drawOval(rect, maskPaint)
   318	    imagePaint.xfermode = PorterDuffXfermode(PorterDuff.Mode.SRC_IN)
   319	    canvas.drawBitmap(source, null, rect, imagePaint)
   320	    imagePaint.xfermode = null
   321	    return output
   322	}

     1	# Cinerific
     2	
     3	Local Android project for the `prahlin/cinerific` GitHub repo.
     4	
     5	This is a first Kotlin / Jetpack Compose scaffold based on the Figma intro and sign-in frames. The current app opens as a tablet-friendly full-screen frame preview with:
     6	
     7	- Intro frame 1
     8	- Animated intro frame 2
     9	- Promo/logo intro frame 3
    10	- Sign-in/loading frame 4
    11	
    12	## Stack
    13	
    14	- Kotlin
    15	- Jetpack Compose Material 3
    16	- Android Gradle Plugin 8.5.x
    17	- Gradle 8.7
    18	
    19	## Run it
    20	
    21	Open the project in Android Studio and run `app`, or use the Gradle wrapper:
    22	
    23	```bash
    24	./gradlew :app:assembleDebug
    25	```
    26	
    27	The project pins Gradle to Android Studio's bundled JBR in `gradle.properties`, so command-line builds do not accidentally use an unsupported system JDK.
    28	
    29	From VS Code, run the default build task `Android: assemble debug`. There are also tasks for `Android: install debug` and `Android: launch Cinerific` when an emulator or device is connected.
    30	
    31	## Notes for the first Kotlin pass
    32	
    33	- The layout is intentionally simple and adaptive for tablets.
    34	- The current visuals are Compose approximations of the Figma frames and are meant to be replaced with production assets screen by screen.
    35	- The local project folder is the same path as the GitHub remote: `prahlin/cinerific`.
