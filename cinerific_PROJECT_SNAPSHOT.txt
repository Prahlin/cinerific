   748	private fun bootProgressAt(bootStartMillis: Long): Float {
   749	    val elapsed = SystemClock.uptimeMillis() - bootStartMillis
   750	    return (elapsed / BOOT_ANIMATION_MS.toFloat()).coerceIn(0f, 1f)

   753	private fun easedSegment(value: Float, start: Float, end: Float): Float {
   754	    val progress = ((value - start) / (end - start)).coerceIn(0f, 1f)
   755	    return FastOutSlowInEasing.transform(progress)
   756	}
   757	
   758	private fun linearSegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   759	    val ms = progress.coerceIn(0f, 1f) * BOOT_ANIMATION_MS
   760	    return ((ms - startMs) / (endMs - startMs).toFloat()).coerceIn(0f, 1f)
   761	}
   762	
   763	private fun easedSegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   764	    return FastOutSlowInEasing.transform(linearSegmentMs(progress, startMs, endMs))
   765	}
   766	
   767	private fun bouncySegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   768	    return bouncyEasing(linearSegmentMs(progress, startMs, endMs))
   769	}
   770	
   771	private fun bouncyEasing(value: Float): Float {
   772	    val t = value.coerceIn(0f, 1f)
   773	    val base = FastOutSlowInEasing.transform(t)
   774	    val bounce = sin(t * PI.toFloat() * 4.5f) * (1f - t) * 0.13f
   775	    return base + bounce
   776	}
   777	
   778	private fun lerpFloat(start: Float, end: Float, amount: Float) = start + (end - start) * amount
   779

   781	    x = lerpFloat(start.x, end.x, amount),

   783	    w = lerpFloat(start.w, end.w, amount),
   784	    h = lerpFloat(start.h, end.h, amount)
   785	)

   256	    val bounds: Bounds
   257	)
   258	
   259	private data class Bounds(
   260	    val x: Float,
   261	    val y: Float,
   262	    val w: Float,
   263	    val h: Float
   264	)
   265	
   266	private fun bootProgressAt(bootStartMillis: Long): Float {
   267	    val elapsed = SystemClock.uptimeMillis() - bootStartMillis
   268	    return (elapsed / BOOT_ANIMATION_MS.toFloat()).coerceIn(0f, 1f)
   269	}
   270	
   271	private fun linearSegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   272	    val ms = progress.coerceIn(0f, 1f) * BOOT_ANIMATION_MS
   273	    return ((ms - startMs) / (endMs - startMs).toFloat()).coerceIn(0f, 1f)
   274	}
   275	
   276	private fun easedSegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   277	    return FastOutSlowInEasing.transform(linearSegmentMs(progress, startMs, endMs))
   278	}
   279	
   280	private fun bouncySegmentMs(progress: Float, startMs: Int, endMs: Int): Float {
   281	    return bouncyEasing(linearSegmentMs(progress, startMs, endMs))
   282	}
   283	
   284	private fun bouncyEasing(value: Float): Float {
   285	    val t = value.coerceIn(0f, 1f)
   286	    val base = FastOutSlowInEasing.transform(t)
   287	    val bounce = sin(t * PI.toFloat() * 4.5f) * (1f - t) * 0.13f
   288	    return base + bounce
   289	}
   290	
   291	private fun lerpFloat(start: Float, end: Float, amount: Float) = start + (end - start) * amount
   292	
   293	private fun lerpBounds(start: Bounds, end: Bounds, amount: Float) = Bounds(
   294	    x = lerpFloat(start.x, end.x, amount),
   295	    y = lerpFloat(start.y, end.y, amount),
   296	    w = lerpFloat(start.w, end.w, amount),
   297	    h = lerpFloat(start.h, end.h, amount)
   298	)
   299	
   300	private fun lerpColor(start: Int, end: Int, amount: Float): Int {
   301	    val t = amount.coerceIn(0f, 1f)
   302	    return Color.argb(
   303	        lerpInt(Color.alpha(start), Color.alpha(end), t),
   304	        lerpInt(Color.red(start), Color.red(end), t),
   305	        lerpInt(Color.green(start), Color.green(end), t),
   306	        lerpInt(Color.blue(start), Color.blue(end), t)
   307	    )
   308	}
   309	
   310	private fun lerpInt(start: Int, end: Int, amount: Float): Int {
   311	    return (start + (end - start) * amount).roundToInt()
   312	}
   313	
   314	private fun combineBitmaps(base: Bitmap, overlay: Bitmap): Bitmap {
   315	    val output = Bitmap.createBitmap(base.width, base.height, Bitmap.Config.ARGB_8888)
   316	    val canvas = Canvas(output)
   317	    val paint = Paint(Paint.ANTI_ALIAS_FLAG or Paint.FILTER_BITMAP_FLAG or Paint.DITHER_FLAG)
   318	    canvas.drawBitmap(base, 0f, 0f, paint)
   319	    canvas.drawBitmap(overlay, 0f, 0f, paint)
   320	    return output
   321	}
   322	
   323	private fun circularBitmap(source: Bitmap): Bitmap {
   324	    val output = Bitmap.createBitmap(source.width, source.height, Bitmap.Config.ARGB_8888)
   325	    val canvas = Canvas(output)
   326	    val rect = RectF(0f, 0f, source.width.toFloat(), source.height.toFloat())
   327	    val maskPaint = Paint(Paint.ANTI_ALIAS_FLAG)
   328	    val imagePaint = Paint(Paint.ANTI_ALIAS_FLAG or Paint.FILTER_BITMAP_FLAG or Paint.DITHER_FLAG)
   329	    canvas.drawOval(rect, maskPaint)
   330	    imagePaint.xfermode = PorterDuffXfermode(PorterDuff.Mode.SRC_IN)
   331	    canvas.drawBitmap(source, null, rect, imagePaint)
   332	    imagePaint.xfermode = null
   333	    return output
   334	}

    61	print_file() {

    63	
    64	  if [ -f "$file" ]; then

    93	  fi

    99	    ! -name "*.keystore" \
   100	    | sort | while IFS= read -r file; do

   102	    local modified

   104	    modified="$(stat -f "%Sm" -t "%Y-%m-%d %H:%M:%S" "$file" 2>/dev/null || stat -c "%y" "$file" 2>/dev/null || echo "unknown")"
   105	    printf '%s | %s bytes | modified %s\n' "${file#$PROJECT_DIR/}" "$size" "$modified" >> "$OUTPUT_FILE"
